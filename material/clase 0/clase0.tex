\section{Metodología}

\begin{frame}{¿Cómo funcionará el curso?}
    
    \begin{orangebox}[Simulación de Entorno Laboral]
        Este NO es un curso tradicional. Es una experiencia práctica simulando una empresa real.
    \end{orangebox}
    
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Equipos de 3 personas:}
            \begin{itemize}
                \item Roles rotativos semanales
                \item \textbf{Tech Lead:} Revisa y aprueba código
                \item \textbf{Developers:} Implementan features
            \end{itemize}
            
            \vspace{0.3cm}
            
            \textbf{Cada sesión:}
            \begin{itemize}
                \item 30-50 min teoría
                \item 60-70 min práctica
                \item Trabajo colaborativo en tiempo real
                \item Resolución inmediata de dudas
            \end{itemize}
        \end{column}
        \pause
        \begin{column}{0.48\textwidth}
            \textbf{Proyecto colaborativo:}
            \begin{itemize}
                \item Stack tecnológico a elección
                \item Desarrollo incremental
                \item Entregas parciales
                \item Pipeline CI/CD funcional
            \end{itemize}
            
            \vspace{0.3cm}
            
            \textbf{Apoyo del GLUD:}
            \begin{itemize}
                \item Plantillas base
                \item Auditoría final
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Stack Tecnológico Flexible}
    
    \begin{orangebox}[Agnóstico del Lenguaje]
        Puedes usar el lenguaje que prefieras. El curso se adapta a tu stack.
    \end{orangebox}
    
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Sugerencia inicial:}
            
            \vspace{0.2cm}
            
            \begin{darkbox}[Java + Maven]
                \centering
                {\small
                Java JDK 17+ \\
                Apache Maven \\
                }
                {\tiny
                \begin{itemize}
                    \item Fácil integración CI/CD
                    \item Comandos estándar (\texttt{mvn test})
                    \item Ecosistema maduro
                \end{itemize}
                }
            \end{darkbox}
        \end{column}
        \pause
        \begin{column}{0.48\textwidth}
            \textbf{Otras opciones válidas:}
            
            \vspace{0.2cm}
            
            {\small
            \begin{itemize}
                \item \textbf{Python} + pytest/unittest
                \item \textbf{JavaScript/TypeScript} + npm/Jest
                \item \textbf{C++} + CMake/Google Test
                \item \textbf{C\#} + .NET
                \item Cualquier lenguaje con tests automatizables
            \end{itemize}
            }
            
            \vspace{0.3cm}
            
            {\color{gludOrangeLight} Lo importante es que tenga build tool y testing framework}
        \end{column}
